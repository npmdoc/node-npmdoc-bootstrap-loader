  // We need to match user loaders in different formats:

      throw new Error('\n        I can\'t find \'sass-loader\'.\n        Add it to array of loaders in .bootstraprc.\n      ');

    } else {

    }

    // eslint-disable-next-line no-param-reassign

  }
